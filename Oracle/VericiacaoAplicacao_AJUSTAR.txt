

SELECT OPEN_MODE FROM V$DATABASE;

SELECT STATUS FROM V$INSTANCE;


SELECT * FROM V$SESSION ;


SELECT  
  JOB_NAME
  ,STATUS
  ,RUN_DURATION
  ,LOG_DATE
  ,REQ_START_DATE
  ,ACTUAL_START_DATE
  ,ADDITIONAL_INFO

  
FROM DBA_SCHEDULER_JOB_RUN_DETAILS
  WHERE OWNER NOT IN ('SYS')
  AND STATUS NOT IN ('SUCCEEDED')
ORDER BY 
  ACTUAL_START_DATE;
  
  
-- Consultar privilégios de sistema 
SELECT 
  PRIVILEGE
FROM DBA_SYS_PRIVS
  WHERE GRANTEE = 'SIGRH'

ORDER BY 1
  
;

-- Consultar privilégios de objetos 
SELECT 
  OWNER
  ,TABLE_NAME
  ,PRIVILEGE
FROM DBA_TAB_PRIVS
  WHERE GRANTEE = 'SIGRH'

;

-- Consultar roles
SELECT 
  GRANTED_ROLE
FROM DBA_ROLE_PRIVS
  WHERE GRANTEE = 'SIGRH'
  
;
